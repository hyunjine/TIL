import sys
from heapq import heappush, heappop

def solution(n, arr):
  heap = []
  heappush(heap, arr[0][1])
  for i in range(1, n):
    start = arr[i][0]
    heappush(heap, arr[i][1])
    if start >= heap[0]:heappop(heap)
  return len(heap)

rl = sys.stdin.readline
n = int(rl())
arr = sorted([[*map(int, rl().split())] for _ in range(n)])
answer = solution(n, arr)
print(answer)
