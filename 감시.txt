import sys
rl = sys.stdin.readline

words = [rl().rstrip() for _ in range(int(rl()))]
answer = 0

def func(word):
  dic = {}
  for i in range(len(word)-1):
    if word[i] != word[i+1]:
      if dic.get(word[i]) == None:
        dic[word[i]] = 1
      else:
        return False
  if dic.get(word[-1]) != None:
    return False
  return True

for word in words:
  if func(word):
    answer += 1
print(answer)
