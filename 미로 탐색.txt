from collections import*
N,M=map(int, input().split())
g=[[*input()] for _ in ' '*N]
q=deque([(0,0)])
D=[(1,0),(-1,0),(0,1),(0,-1)]
while q:
  p=q.popleft()
  for d in D:
    x,y=p[0]+d[0],p[1]+d[1]
    if 0<=x<N and 0<=y<M and g[x][y]=='1':
      g[x][y]=int(g[p[0]][p[1]])+1
      q.append((x,y))
print(g[N-1][M-1])
