from collections import deque
def solution(l, w, truck):
    truck.reverse()
    bridge = deque()
    s = 0
    answer = 0
    while truck:
        if len(bridge) < l:
            if w >= s+truck[-1]:
                v = truck.pop()
                s += v
                bridge.append(v)
            else:
                bridge.append(0)
        if len(bridge) == l:
            s -= bridge.popleft()
        answer += 1
    return answer+l
