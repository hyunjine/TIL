N = int(input())
graph = [[*input()] for _ in ' '*N]
D=[(1,0),(-1,0),(0,1),(0,-1)]
def dfs(graph, x, y, cnt):
  if 0<=x<N and 0<=y<N and graph[x][y]=='1':
    graph[x][y]=1
    cnt.append(1)
    for d in D: dfs(graph, x+d[0], y+d[1], cnt)
    return True
  return False
answer = []
for i in range(N):
  for j in range(N):
    temp = []
    if dfs(graph, i, j, temp): answer.append(sum(temp))

print(len(answer))
for a in sorted(answer):
  print(a)
