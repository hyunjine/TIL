from heapq import heappush, heappop
def solution(scoville, K):
    heap = []
    answer = 0
    while scoville: heappush(heap, scoville.pop())
    while heap and heap[0] < K: 
        answer += 1
        first = heappop(heap)
        if not heap: return -1
        second = heappop(heap)
        heappush(heap, first+(second*2))
    
    return answer
