N = int(input())
M = int(input())
case = [list(map(int, input().split())) for _ in range(M)]
graph = [[] for _ in range(N+1)]
for c in case: 
    graph[c[0]].append(c[1])
    graph[c[1]].append(c[0])
visited = [False]*(N+1)

def dfs(graph, v, visited, cnt):
    visited[v] = True
    for g in graph[v]:
        if not visited[g]:            
            dfs(graph, g, visited, cnt.append(1)):
cnt = []
answer = dfs(graph, 1, visited, cnt)
print(len(cnt))
