def solution(n, lost, reserve):
    set_lost, set_reserve = set(lost), set(reserve)
    LOST = set_lost - set_reserve
    RESERVE = set_reserve - set_lost
    answer = n - len(LOST)
    
    for a in LOST:
        for b in RESERVE:
            if a - 1 == b or a + 1 == b:
                answer += 1
                RESERVE.remove(b)
                break
    return answer
