import java.util.*;
import java.util.stream.*;
class Solution {
    public int solution(String A, String B) {
        int answer = 0;
        Deque<String> a = new ArrayDeque<>(Arrays.asList(A.split("")));
        for (int i = 0; i < A.length(); i++) {
            if (a.stream().collect(Collectors.joining()).equals(B)) return answer;
            answer++;
            String t = a.removeLast();
            a.addFirst(t);
        }
        return -1;
    }
}
