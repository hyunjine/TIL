from collections import deque

def solution(N,M,arr):
  count = 1
  while len(arr) > 1:
    v, i = arr.popleft()
    Max = max(arr)[0]
    if v < Max:
      arr.append((v,i))
    else:
      if i == M:
        return count
      count += 1
  return count
answer = []
for _ in range(int(input())):
  N, M = map(int, input().split())
  arr = deque([(v,i) for i,v in enumerate(list(map(int, input().split())))])
  answer.append(solution(N,M,arr))

for a in answer:
  print(a)
