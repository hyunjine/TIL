N = int(input())
A = sorted(list(map(int, input().split())))
M = int(input())
arr = list(map(int, input().split()))
for a in arr:
  start, end = 0, N-1
  isFind = False
  while end-start > 1:
    aim = (start+end) // 2
    if a in [A[start],A[end],A[aim]]:
      isFind = True
      break
    elif a < A[aim]:
      end = aim
    else:
      start = aim
  print(int(isFind))
