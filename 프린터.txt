from collections import deque
def solution(p, l):
    arr = [(p[i],i) for i in range(len(p))]
    Q = deque(arr)
    answer = 1
    while Q:
        m = max(Q)[0]
        v = Q.popleft()
        if v[0] < m:
            Q.append(v)
        else:
            if v[1] == l: break
            answer += 1
    return answer
