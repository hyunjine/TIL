import java.util.*;
class Solution {
    public int[] solution(int n) {
        List<Integer> answer = new ArrayList<Integer>();
        for (int i = 1; i <= n; i+=2) {
            answer.add(i);
        }
        int[] t = new int[answer.size()];
        for (int i = 0; i < answer.size(); i++) {
            t[i] = answer.get(i);
        }

        return t;
    }
}
