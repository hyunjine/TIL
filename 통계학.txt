import sys
rl = sys.stdin.readline
N = int(rl())
arr = sorted([int(rl()) for _ in range(N)])
dic = {}
for n in arr:
  if dic.get(n) == None:
    dic[n] = 1
  else:
    dic[n] += 1
# 평균
print(round(sum(arr)/N))
# 중앙값
print(arr[N//2])
# 최빈값
a = sorted(dic.items(), key = lambda x: (-x[1], x[0]))
if len(a) == 1 or a[0][1] != a[1][1]:
  print(a[0][0])
else:
  print(a[1][0])
# 범위
print(arr[-1] - arr[0])
