import java.util.*;
class Solution {
    public String solution(String my_string, String letter) {
        StringBuffer answerList = new StringBuffer();
        
        for (int i = 0; i < my_string.length(); i++) {
            String s = Character.toString(my_string.charAt(i));
            if (!s.equals(letter)) answerList.append(s);
        }
        return answerList.toString();
    }
}
