cnt = 0
answer = 0
def recursion(word, arr, at, num):
    global cnt
    global answer
    visit = [False]*5
    if at < 4 and not visit[at]: # 다음 차수 진행
        for a in arr:
            visit[at] = True
            number = num + a
            cnt = cnt + 1
            if word == number:
                answer = cnt
            recursion(word, arr, at+1, num+a)
    else: # 이번 차수 진행
        for a in arr:
            number = num + a
            cnt = cnt + 1
            if word == number:
                answer = cnt
            
def solution(word):
    arr = ['A', 'E', 'I', 'O', 'U']
    recursion(word, arr, 0, "")
    return answer
