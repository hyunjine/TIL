import sys
def solution(n, m):
  answer = 1
  if n <= 1: 
    pass
  elif m < 7:
    if n == 2:
      answer += int((m-1)/2) if m <= 7 else 3
    else:
      answer += m-1 if m <= 4 else 3
  else:
    if n == 2:
      answer += 3
    else:
      answer += 4 + (m - 7)  
        
  return answer

rl = sys.stdin.readline
n,m = map(int, rl().split())
answer = solution(n,m)
print(answer)
