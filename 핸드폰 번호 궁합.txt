arr = [input() for _ in range(2)]
dp = [[0]*i for i in range(2,17)][::-1]
idx = 0
for i in range(8):
  for j in range(2):
    dp[0][idx] = int(arr[j][i])
    idx += 1

for i in range(1,15):
  for j in range(len(dp[i])):
    dp[i][j] = (dp[i-1][j]+dp[i-1][j+1])%10
print(str(dp[-1][0])+str(dp[-1][1]))
