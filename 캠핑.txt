def solution(arr):
  answer = 0
  v = arr[2]
  p = arr[1]
  l = arr[0]
  while True:
    v -= p
    answer += l
    if v < p:
      answer += min(v, l)
      break
  return answer

arr = []
while True:
  a = list(map(int, input().split()))
  if a[0] == 0: break
  arr.append(a)
for i in range(len(arr)):
  answer = solution(arr[i])
  print("Case %d:" %(i+1), answer)
