import java.util.*;
class Solution {
    public int[] solution(int n) {
        List<Integer> answer = new ArrayList<>();
        int t = 2;
        while (n >=t) {
            if (n % t == 0) {
                n /= t;
                answer.add(t);
            } else {
                t++;
            }
        }
        
        return answer.stream().distinct().mapToInt(Integer::intValue).toArray();
    }
}
