from collections import deque

K = int(input())
arr = deque([list(map(int, input().split())) for _ in range(6)])

width = max([a[1] for a in arr if a[0] == 3 or a[0] == 4])
height = max([a[1] for a in arr if a[0] == 1 or a[0] == 2])

temp = [a[0] for a in arr]
temp = [t for t in [1,2,3,4] if temp.count(t) == 2]
while arr[0][0] in temp or arr[1][0] in temp:
  arr.append(arr.popleft())

area = width*height
part = arr[3][1]*arr[4][1]
answer = (area-part)*K
print(answer)
