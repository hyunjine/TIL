import sys

def solution(name, error):
  answer = ''
  center = ''  
  while name:
    s1 = name.pop()
    s2 = ''
    if name: s2 = name.pop()
    if s1 != s2:
      if center: return error
      center = s1
      name.append(s2)
      continue
    else:
      answer += s1
  answer = ''.join(reversed(answer)) + center + answer
  return answer


rl = sys.stdin.readline
name = sorted(list(rl().rstrip()))
error = "I'm Sorry Hansoo"
answer = solution(name, error)
print(answer)
